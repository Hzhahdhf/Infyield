local function applyHover(button, idleColor, hoverColor)
	button.MouseEnter:Connect(function()
		TweenService:Create(button, TweenInfo.new(0.15), { BackgroundColor3 = hoverColor }):Play()

	button.MouseLeave:Connect(function()

	end)

	-- [[ ANTI LAG 2.0: Reduced Lighting Intensity & FX Removal ]]
	local antiLagActive = false
	local antiLagStore = {}
	local antiLagWatcher = nil
	local function antiLagApply(obj)
		if obj:IsA("ParticleEmitter") or obj:IsA("Trail") or obj:IsA("Beam") or obj:IsA("Fire") or obj:IsA("Smoke") or obj:IsA("PostEffect") then
			table.insert(antiLagStore, { obj, "Enabled", obj.Enabled })
			pcall(function() obj.Enabled = false end)
		end
	end
	
	local antiLagRow, antiLagTitle, antiLagStatus = makeFeatureRow(mainPage, "🚀", "Anti Lag", 6)
	local function enableAntiLag()
		antiLagStore = {}
		task.spawn(function()
			for i, obj in ipairs(workspace:GetDescendants()) do
				antiLagApply(obj)
				if i % 40 == 0 then RunService.Heartbeat:Wait() end
			end
		end)
		
		-- Store exactly how lighting was before touching it
		table.insert(antiLagStore, { Lighting, "Ambient", Lighting.Ambient })
		table.insert(antiLagStore, { Lighting, "OutdoorAmbient", Lighting.OutdoorAmbient })
		table.insert(antiLagStore, { Lighting, "Brightness", Lighting.Brightness })
		
		-- Reduce light intensity exactly by 20%
		Lighting.Ambient = Color3.new(Lighting.Ambient.R * 0.8, Lighting.Ambient.G * 0.8, Lighting.Ambient.B * 0.8)
		Lighting.OutdoorAmbient = Color3.new(Lighting.OutdoorAmbient.R * 0.8, Lighting.OutdoorAmbient.G * 0.8, Lighting.OutdoorAmbient.B * 0.8)
		Lighting.Brightness = Lighting.Brightness * 0.8
		
		antiLagWatcher = workspace.DescendantAdded:Connect(function(obj)
			if antiLagActive then task.defer(antiLagApply, obj) end
		end)
		antiLagActive = true
	end
	
	local function disableAntiLag()
		if antiLagWatcher then antiLagWatcher:Disconnect(); antiLagWatcher = nil end
		-- Restore exactly the values we saved
		for i = #antiLagStore, 1, -1 do
			local entry = antiLagStore[i]
			pcall(function() entry[1][entry[2]] = entry[3] end)
		end
		antiLagStore = {}
		antiLagActive = false
	end
	
	antiLagRow.MouseButton1Click:Connect(function()
		if antiLagActive then
			disableAntiLag()
			antiLagRow.BackgroundColor3 = THEME.ButtonIdle
			antiLagStatus.Text = "OFF"
			antiLagStatus.TextColor3 = THEME.TextDim
			showToast("Anti Lag Disabled")
		else
			enableAntiLag()
			antiLagRow.BackgroundColor3 = THEME.ButtonOn
			antiLagStatus.Text = "ON"
			antiLagStatus.TextColor3 = THEME.TextMain
			showToast("Anti Lag Enabled", THEME.ButtonOn)
		end
	end)

local function checkPassword()
	if locked then return end
	if passBox.Text == TARGET_PASSWORD then
		passFrame:Destroy()
		loadMainScript()
	else
		attempts = attempts + 1
		if attempts >= 5 then
			locked = true
			passBox.Text = ""
			passBox.PlaceholderText = "Locked..."
			passBox.Active = false
			passBtn.AutoButtonColor = false
			task.spawn(function()
				for i = 60, 1, -1 do
					passStatus.Text = "Locked. Try again in " .. i .. "s"
					task.wait(1)
				end
				locked = false
				attempts = 0
				passStatus.Text = "Incorrect password."
				passBox.PlaceholderText = "Enter Password..."
				passBox.Active = true
				passBtn.AutoButtonColor = true
			end)
		else
			passStatus.Text = "Incorrect password."
		end
	end
end

passBtn.MouseButton1Click:Connect(checkPassword)
passBox.FocusLost:Connect(function(enterPressed)
	if enterPressed then checkPassword() end
end)
