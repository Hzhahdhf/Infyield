local function corner(parent, radius)
	local c = Instance.new("UICorner")
	c.CornerRadius = UDim.new(0, radius or 5)
	c.Parent = parent

end

local function applyHover(button, idleColor, hoverColor, activeCheck)
	button.MouseEnter:Connect(function()
		if activeCheck and activeCheck() then return end
		TweenService:Create(button, TweenInfo.new(0.15, Enum.EasingStyle.Sine), { BackgroundColor3 = hoverColor }):Play()
	end)
	button.MouseLeave:Connect(function()
		if activeCheck and activeCheck() then return end
		TweenService:Create(button, TweenInfo.new(0.15, Enum.EasingStyle.Sine), { BackgroundColor3 = idleColor }):Play()
	end)
end

	local function runMassiveOptimization()
		for _, v in pairs(Lighting:GetChildren()) do
			if v:IsA("Sky") or v:IsA("Atmosphere") or v:IsA("BloomEffect") or v:IsA("BlurEffect") or v:IsA("ColorCorrectionEffect") or v:IsA("DepthOfFieldEffect") or v:IsA("SunRaysEffect") then
				v:Destroy()
			end
		end
		
		local solidSky = Instance.new("Sky")
		solidSky.SkyboxBk, solidSky.SkyboxDn, solidSky.SkyboxFt, solidSky.SkyboxLf, solidSky.SkyboxRt, solidSky.SkyboxUp = "", "", "", "", "", ""
		solidSky.CelestialBodiesShown = false
		solidSky.Parent = Lighting
		
		Lighting.GlobalShadows = false
		Lighting.FogEnd = 9e9
		Lighting.Brightness = 0
		Lighting.Ambient = Color3.fromRGB(130, 130, 130)
		Lighting.OutdoorAmbient = Color3.fromRGB(130, 130, 130)
		
		local function targetAsset(obj)
			pcall(function()
				if obj:IsA("BasePart") then
					obj.Material = Enum.Material.SmoothPlastic
					obj.Reflectance = 0
					obj.CastShadow = false
				elseif obj:IsA("Decal") or obj:IsA("Texture") or obj:IsA("Clothing") or obj:IsA("ShirtGraphic") then
					obj:Destroy()
				elseif obj:IsA("ParticleEmitter") or obj:IsA("Trail") or obj:IsA("Beam") or obj:IsA("Fire") or obj:IsA("Smoke") or obj:IsA("Sparkles") then
					obj:Destroy()
				elseif obj:IsA("Light") then
					obj.Shadows = false
				elseif obj:IsA("MeshPart") then
					obj.Material = Enum.Material.SmoothPlastic
					obj.CastShadow = false
					obj.TextureID = ""
				elseif obj:IsA("SpecialMesh") then
					obj.TextureId = ""
				end
			end)
		end
		
		for _, obj in pairs(workspace:GetDescendants()) do targetAsset(obj) end
		workspace.DescendantAdded:Connect(targetAsset)
		
		pcall(function()
			settings().Rendering.QualityLevel = Enum.QualityLevel.Level01
			workspace.Terrain.Decoration = false
			workspace.Terrain.WaterWaveSize = 0
			workspace.Terrain.WaterWaveSpeed = 0
			workspace.Terrain.WaterReflectance = 0
			workspace.Terrain.WaterTransparency = 0
		end)
	end

		showToast("Anti Lag Active! Textures purged, shadow logic dropped.", THEME.Success)

	end)

		if isHopping then return end
		isHopping = true
		hopRow.BackgroundColor3 = THEME.ButtonOn
		showToast("Scanning for servers...", THEME.TextMain)
		task.spawn(function()
			local placeId = game.PlaceId
			local url = "https://games.roblox.com/v1/games/" .. placeId .. "/servers/Public?sortOrder=Asc&limit=100"
			local success, result = pcall(function() return game:HttpGet(url) end)
			if success and result then
				local data = HttpService:JSONDecode(result)
				if data and data.data then
					local minPlayers, bestServer = math.huge, nil
					for _, srv in ipairs(data.data) do
						if srv.playing and srv.maxPlayers and srv.playing < srv.maxPlayers and srv.id ~= game.JobId then
							if srv.playing > 0 and srv.playing < minPlayers then
								minPlayers = srv.playing
								bestServer = srv.id
							end
						end
					end
					if bestServer then TeleportService:TeleportToPlaceInstance(placeId, bestServer, player); return end
				end
			end
			showToast("No valid servers found!", THEME.TextMain)
			isHopping = false
			hopRow.BackgroundColor3 = THEME.ButtonIdle
		end)
	end)

	local function teleportToPlayer(targetName)
		while teleporting do
			task.wait(0.1)
			local target = Players:FindFirstChild(targetName)
			if target and target.Character and target.Character:FindFirstChild("HumanoidRootPart") then
				local myChar = player.Character
				local myHRP = myChar and myChar:FindFirstChild("HumanoidRootPart")
				if myHRP then myHRP.CFrame = target.Character.HumanoidRootPart.CFrame + Vector3.new(0, 3, 0) end
			end
		end
	end

	local antiAfkConnection = nil
	local function setAntiAfk(on)
		if on and not antiAfkConnection then
			antiAfkConnection = player.Idled:Connect(function()
				pcall(function()
					VirtualInputManager:SendKeyEvent(true, Enum.KeyCode.Space, false, nil)
					VirtualInputManager:SendKeyEvent(false, Enum.KeyCode.Space, false, nil)
				end)
			end)
		elseif not on and antiAfkConnection then
			antiAfkConnection:Disconnect()
			antiAfkConnection = nil
		end
	end
	if genv.antiAfkOn then setAntiAfk(true) end

		end
	end)
