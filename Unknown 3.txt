local function applyHover(button, idleColor, hoverColor, activeCheck, strokeElement)
	button.MouseEnter:Connect(function()
		if activeCheck and activeCheck() then return end
		TweenService:Create(button, TweenInfo.new(0.3, Enum.EasingStyle.Sine), { BackgroundColor3 = hoverColor }):Play()
		if strokeElement then
			TweenService:Create(strokeElement, TweenInfo.new(0.3, Enum.EasingStyle.Sine), { Color = THEME.Accent }):Play()
		end
	end)
	button.MouseLeave:Connect(function()
		if activeCheck and activeCheck() then return end
		TweenService:Create(button, TweenInfo.new(0.3, Enum.EasingStyle.Sine), { BackgroundColor3 = idleColor }):Play()
		if strokeElement then
			TweenService:Create(strokeElement, TweenInfo.new(0.3, Enum.EasingStyle.Sine), { Color = THEME.AccentLine }):Play()
		end
	end)
end

	local function runMassiveOptimization()
		-- Extreme Settings Overrides
		pcall(function() settings().Rendering.QualityLevel = Enum.QualityLevel.Level01 end)
		pcall(function() settings().Rendering.MeshPartDetailLevel = Enum.MeshPartDetailLevel.Level04 end)
		pcall(function() settings().Physics.PhysicsEnvironmentalThrottle = Enum.EnviromentalPhysicsThrottle.Retrograde end)
		pcall(function() setfpscap(30) end) -- Lowers CPU/GPU usage heavily for AFK grinding
		
		local function purgeLightingFx()
			for _, v in pairs(Lighting:GetChildren()) do
				if v:IsA("Sky") or v:IsA("Atmosphere") or v:IsA("BloomEffect") or v:IsA("BlurEffect") or v:IsA("ColorCorrectionEffect") or v:IsA("DepthOfFieldEffect") or v:IsA("SunRaysEffect") or v:IsA("PostEffect") then
					v:Destroy()
				end
			end
		end
		purgeLightingFx()
		Lighting.DescendantAdded:Connect(function(v) task.defer(purgeLightingFx) end)
		
		pcall(function() Lighting.Technology = Enum.Technology.Compatibility end)
		Lighting.GlobalShadows = false
		Lighting.FogEnd = 9e9
		Lighting.Brightness = 0
		Lighting.Ambient = Color3.fromRGB(140, 140, 140)
		Lighting.OutdoorAmbient = Color3.fromRGB(140, 140, 140)
		
		local terrain = workspace:FindFirstChildOfClass("Terrain")
		if terrain then
			pcall(function()
				terrain.WaterWaveSize = 0; terrain.WaterWaveSpeed = 0; terrain.WaterReflectance = 0; terrain.WaterTransparency = 0; terrain.Decoration = false
			end)
		end

		local function targetAsset(obj)
			pcall(function()
				if obj:IsA("BasePart") then
					obj.Material = Enum.Material.SmoothPlastic
					obj.Reflectance = 0
					obj.CastShadow = false
					if obj:IsA("MeshPart") then 
						obj.TextureID = ""
						obj.RenderFidelity = Enum.RenderFidelity.Performance 
					end
				elseif obj:IsA("SpecialMesh") then 
					obj.TextureId = ""
				elseif obj:IsA("Decal") or obj:IsA("Texture") or obj:IsA("Clothing") or obj:IsA("ShirtGraphic") or obj:IsA("Shirt") or obj:IsA("Pants") then 
					obj:Destroy()
				elseif obj:IsA("ParticleEmitter") or obj:IsA("Trail") or obj:IsA("Beam") or obj:IsA("Fire") or obj:IsA("Smoke") or obj:IsA("Sparkles") or obj:IsA("Explosion") then 
					obj:Destroy()
				elseif obj:IsA("Light") then 
					obj.Enabled = false
				elseif obj:IsA("Highlight") or obj:IsA("PostEffect") then
					obj:Destroy()
				end
			end)
		end
		for _, obj in pairs(workspace:GetDescendants()) do targetAsset(obj) end
		workspace.DescendantAdded:Connect(targetAsset)
	end
